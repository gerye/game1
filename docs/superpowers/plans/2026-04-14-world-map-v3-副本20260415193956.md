- [ ] **Step 2：在 `src/world-map.js` 末尾添加 `buildCityTerritories()` 函数**

```javascript
/**
 * 预计算城市领地（Voronoi）：每个地图格子归属于距离最近的城池
 * 返回 { [cityId]: string[] } 其中 string[] 是 "q,r" 格式的 hex key 列表
 * 一次性在 createWorldState() 时调用，结果存入 worldState.cityTerritories
 */
export function buildCityTerritories() {
  const cityPositions = ALL_CITIES.map((c) => ({ id: c.id, q: c.q, r: c.r }));
  const territories = Object.fromEntries(ALL_CITIES.map((c) => [c.id, []]));

  for (let q = -WORLD_MAP_RADIUS; q <= WORLD_MAP_RADIUS; q++) {
    for (let r = -WORLD_MAP_RADIUS; r <= WORLD_MAP_RADIUS; r++) {
      if (!inBounds(q, r)) continue;
      let nearestId = null;
      let nearestDist = Infinity;
      for (const city of cityPositions) {
        const d = hexDistance(q, r, city.q, city.r);
        if (d < nearestDist) { nearestDist = d; nearestId = city.id; }
      }
      if (nearestId) territories[nearestId].push(`${q},${r}`);
    }
  }
  return territories;
}
```

## Task 4：攻城 AI（门派扩张决策）

**Files:**
- Modify: `src/world-events.js`

攻城 AI 策略：每个门派每时节可发起最多 1 次攻城。
评分规则：
- 候选城市 = 当前门派领土内或相邻的中立城市、或相邻的敌方城市
- 分数 = `defensibilityScore` - `riskScore`
- `defensibilityScore` = 候选城市到最近敌方 HQ 的距离（越远越安全）
- `riskScore` = 0（中立城市）或 10（已有敌方兵力的城市）
- 只攻打中立城市（声望消耗低），除非门派声望充足才考虑攻击敌方城市

- [ ] **Step 1：在 `src/world-events.js` 中添加 `runSiegeAI` 函数**

在文件末尾（`checkAndTriggerLastStand` 之后）添加：

```javascript
// ── 攻城 AI ──────────────────────────────────────

/**
 * 判断 cityId 是否与某门派已控制领地相邻
 * （领地相邻 = 该城池的 Voronoi 区域与门派控制的某城市区域存在相邻六格）
 * 简化判断：该城池模板坐标与门派控制的某城池坐标距离 ≤ 某阈值
 */
function isCityReachable(cityId, factionId, worldState) {
  const template = ALL_CITIES.find((c) => c.id === cityId);
  if (!template) return false;

  const ownedCityIds = new Set(
    worldState.cities.filter((c) => c.faction === factionId).map((c) => c.id)
  );

  for (const ownedId of ownedCityIds) {
    const ownedTemplate = ALL_CITIES.find((c) => c.id === ownedId);
    if (!ownedTemplate) continue;
    // 两座城市坐标距离 ≤ 14 格视为可达（考虑大城领地覆盖半径）
    const { hexDistance: hd } = await import("./world-map.js");  // 不能用 await import 在同步函数
    // 直接引用已导入的 hexDistance
    if (hexDistance(template.q, template.r, ownedTemplate.q, ownedTemplate.r) <= 14) {
      return true;
    }
  }
  return false;
}

/**
 * 为单个门派选择最优攻城目标
 * 返回 cityId 或 null
 */
function chooseSiegeTarget(factionId, worldState) {
  const ownCityIds = new Set(
    worldState.cities.filter((c) => c.faction === factionId).map((c) => c.id)
  );

  // HQ 位置 map（用于计算防御价值）
  const hqPositions = {};
  for (const city of ALL_CITIES) {
    if (city.tier === WORLD_CITY_TIERS.HQ) {
      hqPositions[city.id.replace(/-hq$/, "")] = { q: city.q, r: city.r };
    }
  }

  // 候选：未被本门派控制的城市，且可达
  const candidates = ALL_CITIES.filter((template) => {
    const currentState = worldState.cities.find((c) => c.id === template.id);
    if (!currentState) return false;
    if (currentState.faction === factionId) return false; // 已控制

    // 可达判断：与本门派任意已控制城市距离 ≤ 14
    for (const ownedId of ownCityIds) {
      const ownedTemplate = ALL_CITIES.find((c) => c.id === ownedId);
      if (!ownedTemplate) continue;
      if (hexDistance(template.q, template.r, ownedTemplate.q, ownedTemplate.r) <= 14) {
        return true;
      }
    }
    return false;
  });

  if (!candidates.length) return null;

  // 评分：优先中立城市，同等情况下选离敌方 HQ 最远的
  let bestCity = null;
  let bestScore = -Infinity;

  for (const candidate of candidates) {
    const currentState = worldState.cities.find((c) => c.id === candidate.id);
    const isNeutral = !currentState?.faction;

    // 只在声望充足时攻打有主城市（简化：声望 >= 200）
    if (!isNeutral) {
      const prestige = worldState.factionStats[factionId]?.prestige || 0;
      if (prestige < 200) continue;
    }

    // 防御价值 = 到最近敌方 HQ 的距离（越大越好）
    let minEnemyHQDist = Infinity;
    for (const [hqFaction, pos] of Object.entries(hqPositions)) {
      if (hqFaction === factionId) continue;
      const d = hexDistance(candidate.q, candidate.r, pos.q, pos.r);
      if (d < minEnemyHQDist) minEnemyHQDist = d;
    }

    const score = (isNeutral ? 50 : 0) + minEnemyHQDist;
    if (score > bestScore) {
      bestScore = score;
      bestCity = candidate.id;
    }
  }

  return bestCity;
}

/**
 * 运行所有门派的攻城 AI，返回 { worldState, siegeEvents }
 * siegeEvents: [ { factionId, cityId, cityName } ]（已扣减声望，待执行战斗）
 */
export function runSiegeAI(worldState) {
  let ws = worldState;
  const siegeEvents = [];

  for (const factionId of FACTION_IDS) {
    const targetCityId = chooseSiegeTarget(factionId, ws);
    if (!targetCityId) continue;

    const { worldState: newWs, launched } = tryLaunchSiege(ws, factionId, targetCityId);
    if (launched) {
      ws = newWs;
      const template = ALL_CITIES.find((c) => c.id === targetCityId);
      siegeEvents.push({ factionId, cityId: targetCityId, cityName: template?.name || targetCityId });
    }
  }

  return { worldState: ws, siegeEvents };
}
```

```javascript
/**
 * 推进一个时节
 * @param {Object} worldState
 * @param {Object[]} builds
 * @param {Function|null} fastSimFn  可选，签名 (buildA, buildB) => {winnerBuildId, loserBuildId}
 *   传入时会执行相邻单挑；不传时跳过单挑（用于快速推演）
 * @returns {{ worldState, duelResults }}
 *   duelResults: [ { attackerBuildId, defenderBuildId, winnerBuildId } ]
 */
export function advanceSeason(worldState, builds, fastSimFn = null) {
  let ws = { ...worldState, season: worldState.season + 1 };

  // 1. 金币产出
  ws = { ...ws, factionStats: tickGoldProduction(ws.factionStats, ws.cities) };

  // 2. 解除到期重伤
  const newCharStates = { ...ws.characterStates };
  builds.forEach((build) => {
    const cs = newCharStates[build.buildId];
    if (!cs) return;
    if (cs.injured && cs.injuredUntilSeason <= ws.season) {
      newCharStates[build.buildId] = { ...cs, injured: false };
    }
  });
  ws = { ...ws, characterStates: newCharStates };

  // 3. AI 角色状态切换
  ws = aiUpdateCharacterStates(ws, builds);

  // 4. 漫游小事件
  ws = triggerRoamingEvents(ws, builds);

  // 5. 相邻单挑（仅当 fastSimFn 传入时执行）
  const duelResults = [];
  if (fastSimFn) {
    const duels = detectAdjacentDuels(ws, builds);
    for (const { buildA, buildB } of duels) {
      try {
        const result = fastSimFn(buildA, buildB);
        duelResults.push(result);
        if (result.winnerBuildId) {
          ws = addWorldLog(ws, `单挑：${buildA.name || buildA.buildId} vs ${buildB.name || buildB.buildId}，${result.winnerBuildId === buildA.buildId ? (buildA.name || "攻") : (buildB.name || "守")} 胜。`);
        }
      } catch (_) { /* 忽略单挑错误 */ }
    }
  }

  // 6. 攻城 AI（仅内部攻城，不触发真实战场）
  const { worldState: wsAfterSiege, siegeEvents } = runSiegeAI(ws);
  ws = wsAfterSiege;
  // 攻城简化处理：中立城市直接占领；有主城市50%胜率
  for (const evt of siegeEvents) {
    const currentOwner = ws.cities.find((c) => c.id === evt.cityId)?.faction;
    if (!currentOwner) {
      // 中立：直接占领
      ws = applySiegeResult(ws, evt.cityId, evt.factionId, evt.factionId);
    } else {
      // 有主：随机决定（快速模拟）
      const rand = seededRand(evt.factionId + evt.cityId + ws.season);
      const winner = rand > 0.45 ? evt.factionId : currentOwner;
      ws = applySiegeResult(ws, evt.cityId, winner, evt.factionId);
    }
  }

  ws = addWorldLog(ws, `${getSeasonLabel(ws.season)}：各派积蓄力量。`);
  return { worldState: ws, duelResults };
}
```

```javascript
/**
 * 检测大地图上相邻的敌对角色，返回可能发生单挑的配对
 * @returns {{ buildA, buildB }[]}
 */
export function detectAdjacentDuels(worldState, builds) {
  const charStates = worldState.characterStates || {};
  const cityMap = Object.fromEntries(
    ALL_CITIES.map((c) => [c.id, { q: c.q, r: c.r }])
  );

  // 计算每个角色的当前地图坐标
  function getCharPos(build) {
    const cs = charStates[build.buildId];
    if (!cs) return null;
    if (cs.state === WORLD_CHARACTER_STATES.GARRISON && cs.cityId) {
      return cityMap[cs.cityId] || null;
    }
    if (cs.q != null && cs.r != null) return { q: cs.q, r: cs.r };
    return null;
  }

  const activePairs = [];
  const paired = new Set();

  for (let i = 0; i < builds.length; i++) {
    for (let j = i + 1; j < builds.length; j++) {
      const a = builds[i];
      const b = builds[j];
      // 同门派不单挑
      const aKey = a.faction?.key || a.faction;
      const bKey = b.faction?.key || b.faction;
      if (aKey === bKey) continue;
      // 已重伤不单挑
      const csA = charStates[a.buildId];
      const csB = charStates[b.buildId];
      if (csA?.injured || csB?.injured) continue;
      if (paired.has(a.buildId) || paired.has(b.buildId)) continue;

      const posA = getCharPos(a);
      const posB = getCharPos(b);
      if (!posA || !posB) continue;

      const dist = hexDistance(posA.q, posA.r, posB.q, posB.r);
      if (dist <= 3) {
        activePairs.push({ buildA: a, buildB: b });
        paired.add(a.buildId);
        paired.add(b.buildId);
      }
    }
  }

  return activePairs;
}
```

```javascript
function fastPickChaosWinner(entries) {
  const ready = entries.filter((e) => e.build && e.progress);
  if (!ready.length) return null;
  // 按门派统计总实力，随机加权选冠军
  const factionScores = {};
  for (const e of ready) {
    const key = e.build.faction?.key || e.build.faction;
    if (!key) continue;
    const score = (e.progress.level || 1) + Math.random() * 5;
    factionScores[key] = (factionScores[key] || 0) + score;
  }
  const entries2 = Object.entries(factionScores);
  if (!entries2.length) return null;
  entries2.sort((a, b) => b[1] - a[1]);
  return entries2[0][0];
}
```
